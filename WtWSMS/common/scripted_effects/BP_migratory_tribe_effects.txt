create_migratory_tribe_title_tooltip_effect = {
	show_as_tooltip = {
		custom_tooltip = become_migratory_tribe_tt
		if = {
			limit = { exists = scope:new_migratory_tribe_leader_title }
			show_as_tooltip = { get_title = scope:new_migratory_tribe_leader_title }
		}
	}
}

BP_create_migratory_tribe_effect = {
	hidden_effect = {
		#Update all the migration effects
		BP_apply_migration_variables_effect = { CHAR = $CHAR$ }
		
		#Apply all the changes to the counties effected
		BP_apply_county_changes_effect = { CHAR = $CHAR$ }
		
		BP_liberate_existing_vassals = { CHAR = $CHAR$ }
		
		BP_create_migratory_tribe_title_effect = { CHAR = $CHAR$ }
		
		# Transfer landed titles to a new single holder
		BP_transfer_titles_to_single_holder_effect = { CHAR = $CHAR$ }
		
		#Shatter old Realm
		BP_shatter_realm_effect = { INTERITOR = scope:land_inheritor }
		
		#Setup Domicile and Armies & resources
		BP_setup_migratory_domicile_effect = { CHAR = $CHAR$ }
		BP_create_migratory_armies_effect = { CHAR = $CHAR$ }
		BP_get_resources_effect = { CHAR = $CHAR$ }
		
		#Clear variables
		BP_clear_migation_variables_effect = { CHAR = $CHAR$ }
	}
}
BP_apply_migration_variables_effect = {
	$CHAR$ = {
		save_scope_as = migrator
		switch = { #Track which type of tribe to make
			trigger = primary_title.tier
			tier_county = { $CHAR$ = { set_variable = { name = title_tier_county } } }
			tier_duchy = { $CHAR$ = { set_variable = { name = title_tier_duchy } } }
			tier_kingdom = { $CHAR$ = { set_variable = { name = title_tier_kingdom } } }
			tier_empire = { $CHAR$ = { set_variable = { name = title_tier_empire } } }
			fallback = { $CHAR$ = { set_variable = { name = title_tier_county } } }
		}
		set_variable = { name = migration_gold value = migration_gold_to_extract_value }
		set_variable = { name = migration_maa_pool value = migration_population_availability_value }
		this.capital_county = { save_scope_as = old_capital }
	}
}
BP_apply_county_changes_effect = {
	$CHAR$ = {
		every_held_title = {
			limit = {
				this.tier = tier_county
				OR = {
					this.culture = prev.culture
					this.faith = prev.faith
				}
			}
			add_county_modifier = { modifier = county_migrated_away_all_modifier years = migration_time_to_denude }
			#Degrade Development
			if = {
				limit = { this.culture = prev.culture }
				change_development_level = negative_one_forth_development_level
			}
			if = {
				limit = { this.faith = prev.faith }
				change_development_level = negative_one_forth_development_level
			}
		}
	}
}
BP_liberate_existing_vassals = {
	$CHAR$ = {
		create_title_and_vassal_change = {
			type = independency
			save_scope_as = change_releaseVassals
			add_claim_on_loss = no
		}
		every_vassal = {
			limit = { this.primary_title.tier >= tier_county }
			becomes_independent = { change = scope:change_releaseVassals }
		}
		resolve_title_and_vassal_change = scope:change_releaseVassals
	}
}
create_migratory_title_by_tier_effect = {
	if = {
		limit = { this.primary_title = { is_titular = no } }
		create_dynamic_title = {
			tier = $TIER$
			name = migratory_tribe_name_001
			adj = migratory_tribe_name_001_adj
		}
	}
	else = {
		$CHAR$ = { save_scope_as = the_char }
		create_dynamic_title = {
			tier = $TIER$
			name = OLD_TRIBE_NAME
			adj = OLD_TRIBE_ADJ
		}
	}
}
BP_create_migratory_tribe_title_effect = {
	#Actual title creation management
	$CHAR$ = {
		#Create the new title
		if = {
			limit = {
				OR = {
					has_variable = title_tier_kingdom
					has_variable = title_tier_empire
				}
			}
			create_migratory_title_by_tier_effect = {
				TIER = kingdom
				CHAR = $CHAR$
			}
		}
		else = {
			create_migratory_title_by_tier_effect = {
				TIER = duchy
				CHAR = $CHAR$
			}
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change_createNewTitle
			add_claim_on_loss = no
		}
		if = {
			limit = { this.primary_title = { is_titular = no } }
			$CHAR$.house = { save_scope_as = coa_target }
		}
		else = { this.primary_title = { save_scope_as = coa_target } }
		scope:new_title = {
			set_capital_county = scope:old_capital
			set_landless_title = yes
			set_coa = scope:coa_target
			set_destroy_if_invalid_heir = yes
			set_destroy_on_gain_same_tier = yes
			set_delete_on_destroy = yes
			set_no_automatic_claims = yes
			set_definitive_form = yes
			set_can_be_named_after_dynasty = no
			
			change_title_holder = {
				holder = scope:migrator
				change = scope:change_createNewTitle
			}
		}
		resolve_title_and_vassal_change = scope:change_createNewTitle
		
		if = {
			limit = { this.primary_title = { is_titular = yes } }
			scope:new_title = { copy_title_history = $CHAR$.primary_title }
		}
		
		change_government = migrating_tribe_government
	}
}
BP_transfer_titles_to_single_holder_effect = {
	$CHAR$ = {
		every_held_title = {
			limit = {
				is_landless_type_title = no
				tier > tier_barony
			}
			add_to_list = target_titles
		}
		create_title_and_vassal_change = {
			type = conquest
			save_scope_as = change_handOverOldLand
			add_claim_on_loss = no
		}
		every_in_list = {
			list = target_titles
			change_title_holder_include_vassals = {
				holder = scope:land_inheritor
				change = scope:change_handOverOldLand
			}
		}
		resolve_title_and_vassal_change = scope:change_handOverOldLand
	}
}
BP_shatter_realm_effect = {
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change_shatterRealm
		add_claim_on_loss = no
	}
	$INTERITOR$ = {
		add_to_list = new_chars
		#Remove all titular titles held (if possible; not currently possible in 1.14.2.2, as set_delete_on_destroy cannot apply to non-dynamic titles)
		every_held_title = {
			limit = { tier >= tier_duchy }
			prev = { destroy_title = prev }
		}
		every_held_title = {
			limit = {
				this.tier = tier_county
				NOT = { this = prev.capital_county } #Don't take the old inheritor's capital
			}
			this = { save_temporary_scope_as = the_county }
			create_character = {
				template = laamp_inheritor_template
				culture = scope:the_county.culture
				faith = scope:the_county.faith
				dynasty = generate
				location = scope:the_county.title_province
				after_creation = {
					save_scope_as = new_holder
					add_to_list = new_chars
					#Make them relatively obscure peasants
					add_prestige_level = -1
				}
			}
			change_title_holder = {
				holder = scope:new_holder
				change = scope:change_shatterRealm
			}
		}
	}
	every_in_list = { #Everybody has a family to start
		list = new_chars
		trigger_event = game_rule.1001
		remove_from_list = new_chars
	}
	resolve_title_and_vassal_change = scope:change_shatterRealm
}
BP_setup_migratory_domicile_effect = {
	$CHAR$ = {
		if = {
			limit = { has_variable = title_tier_county }
			this.domicile = {
				add_domicile_building = supply_tent_01
				add_domicile_building = baggage_train_01
			}
		}
		else_if = {
			limit = { has_variable = title_tier_duchy }
			this.domicile = {
				add_domicile_building = camp_main_02
				add_domicile_building = supply_tent_01
				add_domicile_building = baggage_train_01
			}
		}
		else_if = {
			limit = { has_variable = title_tier_kingdom }
			this.domicile = {
				add_domicile_building = camp_main_02
				add_domicile_building = camp_main_03
				add_domicile_building = supply_tent_01
				add_domicile_building = supply_tent_02
				add_domicile_building = baggage_train_01
				add_domicile_building = baggage_train_02
			}
		}
		else_if = {
			limit = { has_variable = title_tier_empire }
			this.domicile = {
				add_domicile_building = camp_main_01
				add_domicile_building = camp_main_02
				add_domicile_building = camp_main_03
				add_domicile_building = camp_main_04
				add_domicile_building = supply_tent_01
				add_domicile_building = supply_tent_02
				add_domicile_building = baggage_train_01
				add_domicile_building = baggage_train_02
			}
		}
		else = {
			this.domicile = {
				add_domicile_building = supply_tent_01
				add_domicile_building = baggage_train_01
			}
		}
		this.domicile = { change_provisions = starting_provisions_value }
	}
}
add_regiment_of_type_effect = {
	create_maa_regiment = {
		type = $TYPE$
		check_can_recruit = yes
		size = 1
	}
}
get_a_random_maa_effect = {
	random_list = {
		10 = { add_regiment_of_type_effect = { TYPE = light_footmen } }
		10 = { add_regiment_of_type_effect = { TYPE = bowmen } }
		10 = { add_regiment_of_type_effect = { TYPE = pikemen_unit } }
		10 = {
			trigger = { this.culture = { has_innovation = innovation_quilted_armor } }
			add_regiment_of_type_effect = { TYPE = armored_footmen }
		}
		10 = {
			add_regiment_of_type_effect = { TYPE = light_horsemen }
			modifier = { #Historical Cavalrymen get more horses
				OR = {
					this.culture = { has_cultural_pillar = heritage_sarmatian }
					this.culture = { has_cultural_pillar = heritage_sakan }
					this.culture = { has_cultural_pillar = heritage_turkic }
					this.culture = { has_cultural_pillar = heritage_mongolic }
					this.culture = { has_cultural_pillar = heritage_yugrian }
				}
				factor = 3
			}
			modifier = { #Goths get a bump, Chernyakov's you know...
				this.culture = { the_culture_or_parent_is_with_heritage_trigger = { CULTURE = culture:gothic } }
				factor = 1.5
			}
		}
		60 = {
			trigger = { culture = { has_cultural_parameter = unlock_maa_horse_archers } }
			add_regiment_of_type_effect = { TYPE = horse_archers }
			modifier = {
				number_maa_regiments_of_type = {
					type = horse_archers
					value > 1
				}
				factor = 0.5
			}
			modifier = {
				number_maa_regiments_of_type = {
					type = horse_archers
					value > 2
				}
				factor = 0.5
			}
		}
	}
}
BP_create_migratory_armies_effect = {
	$CHAR$ = {
		#Clear the old regiments, if any
		every_maa_regiment = { destroy_maa_regiment = yes }
		#Create a new list of regiments
		if = {
			limit = { has_variable = title_tier_county }
			get_a_random_maa_effect  = yes
		}
		else_if = {
			limit = { has_variable = title_tier_duchy }
			get_a_random_maa_effect  = yes
			get_a_random_maa_effect  = yes
		}
		else_if = {
			limit = { has_variable = title_tier_kingdom }
			get_a_random_maa_effect  = yes
			get_a_random_maa_effect  = yes
			get_a_random_maa_effect  = yes
		}
		else_if = {
			limit = { has_variable = title_tier_empire }
			get_a_random_maa_effect  = yes
			get_a_random_maa_effect  = yes
			get_a_random_maa_effect  = yes
			get_a_random_maa_effect  = yes
		}
		else = { get_a_random_maa_effect  = yes }
		#Fill them up
		every_maa_regiment = { change_maa_regiment_size = { size = 2 reinforce = yes } }
	}
}
BP_get_resources_effect = {
	$CHAR$ = {
		save_scope_as = the_char
		add_gold = this.var:migration_gold #1.14.2.2: Spams error log but works
		domicile = { change_provisions = prev.var:migration_maa_pool }
	}
}
BP_clear_migation_variables_effect = {
	$CHAR$ = {
		cleanup_variable_effect = { VAR = title_tier_county }
		cleanup_variable_effect = { VAR = title_tier_duchy }
		cleanup_variable_effect = { VAR = title_tier_kingdom }
		cleanup_variable_effect = { VAR = title_tier_empire }
	}
}
