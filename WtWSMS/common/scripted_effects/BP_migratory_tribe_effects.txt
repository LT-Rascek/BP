create_migratory_tribe_title_tooltip_effect = {
	show_as_tooltip = {
		custom_tooltip = become_migratory_tribe_tt
		if = {
			limit = { exists = scope:new_migratory_tribe_leader_title }
			show_as_tooltip = { get_title = scope:new_migratory_tribe_leader_title }
		}
	}
}

#Carbon copy of Vanilla's create_landless_adventurer_title_effect without messaging
BP_create_landless_adventurer_title_effect = {
	$CHAR$ = {
		save_scope_as = migrator
		#Create the new title
		create_dynamic_title = {
			tier = kingdom
			name = migratory_tribe_name_001
			adj = migratory_tribe_name_001_adj
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change1
			add_claim_on_loss = no
		}
		scope:new_title = {
			set_capital_county = title:c_roma
			set_landless_title = yes
			set_coa = title:e_wre
			set_destroy_if_invalid_heir = yes
			set_destroy_on_gain_same_tier = yes
			set_delete_on_destroy = yes
			set_no_automatic_claims = yes
			set_definitive_form = yes
			set_can_be_named_after_dynasty = no
			change_title_holder = {
				holder = scope:migrator
				change = scope:change1
			}
		}
		
		#Hand over old titles
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change2
			add_claim_on_loss = no
		}
		this = {
			every_held_title = {
				limit = { this.tier = tier_county }
				create_character = {
					template = laamp_inheritor_template
					culture = root.capital_county.culture
					faith = root.capital_county.faith
					dynasty = generate
					location = root.capital_province
					after_creation = { save_scope_as = new_holder }
				}
				change_title_holder = {
					holder = scope:new_holder
					change = scope:change2
				}
			}
		}
		
		resolve_title_and_vassal_change = scope:change1
		resolve_title_and_vassal_change = scope:change2
		scope:migrator = { change_government = migrating_tribe_government }
	}
}
