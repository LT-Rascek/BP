convert_adventurer_to_landless_tribe_effect = {
	$CHAR$ = {
		save_scope_as = migrator
		if = {
			limit = { NOT = { has_government = migrating_tribe_government } }
			change_government = migrating_tribe_government
		}
		if = {
			limit = { NOT = { has_realm_law = confederate_partition_succession_law } }
			add_realm_law = confederate_partition_succession_law
		}
		remove_all_realm_laws_effect = yes
		#Destroy the old duchy/kingdom
		scope:land_inheritor = {
			every_held_title = {
				limit = { this.tier = tier_duchy }
				prev = { destroy_title = prev }
			}
		}
		#Create new holders for all the counties
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		scope:land_inheritor = {
			every_held_title = {
				limit = {
					this.tier = tier_county
					NOT = { this = prev.capital_county }
				}
				create_character = {
					template = laamp_inheritor_template
					culture = root.capital_county.culture
					faith = root.capital_county.faith
					dynasty = generate
					location = root.capital_province
					after_creation = { save_scope_as = new_holder }
				}
				change_title_holder = {
					holder = scope:new_holder
					change = scope:change
				}
			}
		}
		resolve_title_and_vassal_change = scope:change
		#Create courtiers if needed
		#Clean up variables that were attached
		primary_title = {
			remove_variable = adventurer_creation_reason
			remove_variable = adventurer_flavor_char
		}
		#Make announcements
		every_player = {
			limit = {
				NOR = {
					root ?= this
					this = scope:migrator
				}
				OR = {
					# Either they're important to us.
					is_close_or_extended_family_of = scope:migrator
					is_consort_of = scope:migrator
					has_important_relationship_with_character_trigger = { CHARACTER = scope:migrator }
					# Or they're just close enough.
					AND = {
						capital_province = { "squared_distance(scope:migrator.location)" <= squared_distance_medium }
						OR = {
							top_liege = root.top_liege
							any_neighboring_and_across_water_top_liege_realm_owner = { this = root.top_liege }
						}
					}
				}
			}
			send_interface_message = {
				type = msg_landless_adventurer_nearby
				title = nearby_migratory_tribe_created_message_title
				left_icon = scope:migrator
				custom_tooltip = nearby_migratory_tribe_created_message_fallback_tt
			}
		}
	}
}
