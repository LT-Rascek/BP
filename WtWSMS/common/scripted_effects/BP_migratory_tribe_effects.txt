create_migratory_tribe_title_tooltip_effect = {
	show_as_tooltip = {
		custom_tooltip = become_migratory_tribe_tt
		if = {
			limit = { exists = scope:new_migratory_tribe_leader_title }
			show_as_tooltip = { get_title = scope:new_migratory_tribe_leader_title }
		}
	}
}

#Carbon copy of Vanilla's create_landless_adventurer_title_effect without messaging
BP_create_migratory_tribe_title_effect = {
	$CHAR$ = {
		save_scope_as = migrator
		switch = { #Track which type of tribe to make
			trigger = primary_title.tier
			tier_county = {}
			tier_duchy = {}
			tier_kingdom = { $CHAR$ = { set_variable = { name = title_tier_kingdom } } }
			tier_empire = { $CHAR$ = { set_variable = { name = title_tier_kingdom} } }
			fallback = {}
		}
		#Apply the appropriate county modifiers
		every_held_title = {
			limit = {
				this.tier = tier_county
				OR = {
					this.culture = prev.culture
					this.faith = prev.faith
				}
			}
			add_county_modifier = { modifier = county_migrated_away_all_modifier years = root.migration_time_to_denude }
		}
	}
	$CHAR$.capital_county = { save_scope_as = old_capital }
	# Transfer landed titles to a new single holder
	every_held_title = {
		limit = {
			is_landless_type_title = no
			tier > tier_barony
		}
		add_to_list = target_titles
	}
	create_title_and_vassal_change = {
		type = conquest
		save_scope_as = change_handOverOldLand
		add_claim_on_loss = no
	}
	hidden_effect = {
		every_in_list = {
			list = target_titles
			change_title_holder_include_vassals = {
				holder = scope:land_inheritor
				change = scope:change_handOverOldLand
			}
		}
		resolve_title_and_vassal_change = scope:change_handOverOldLand
	}
	#Shatter old Realm
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change_shatterRealm
		add_claim_on_loss = no
	}
	scope:land_inheritor = {
		add_to_list = new_chars
		every_held_title = {
			limit = { tier >= tier_duchy }
			prev = { destroy_title = prev }
		}
		every_held_title = {
			limit = {
				this.tier = tier_county
				NOT = { this = prev.capital_county } #Don't take the old inheritor's capital
			}
			this = { save_temporary_scope_as = the_county }
			create_character = {
				template = laamp_inheritor_template
				culture = scope:the_county.culture
				faith = scope:the_county.faith
				dynasty = generate
				location = scope:the_county.title_province
				after_creation = {
					save_scope_as = new_holder
					add_to_list = new_chars
					#Make them relatively obscure peasants
					add_prestige_level = -1
				}
			}
			change_title_holder = {
				holder = scope:new_holder
				change = scope:change_shatterRealm
			}
		}
	}
	every_in_list = { #Everybody has a family to start
		list = new_chars
		trigger_event = game_rule.1001
		remove_from_list = new_chars
	}
	resolve_title_and_vassal_change = scope:change_shatterRealm
	
	#Create new title
	$CHAR$ = {
		#Create the new title
		if = {
			limit = { has_variable = title_tier_kingdom }
			create_dynamic_title = {
				tier = kingdom
				name = migratory_tribe_name_001
				adj = migratory_tribe_name_001_adj
			}
			remove_variable = title_tier_kingdom
		}
		else = {
			create_dynamic_title = {
				tier = duchy
				name = migratory_tribe_name_001
				adj = migratory_tribe_name_001_adj
			}
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change_createNewTitle
			add_claim_on_loss = no
		}
		scope:new_title = {
			set_capital_county = scope:old_capital
			set_landless_title = yes
			set_coa = $CHAR$.house
			set_destroy_if_invalid_heir = yes
			set_destroy_on_gain_same_tier = yes
			set_delete_on_destroy = yes
			set_no_automatic_claims = yes
			set_definitive_form = yes
			set_can_be_named_after_dynasty = no
			change_title_holder = {
				holder = scope:migrator
				change = scope:change_createNewTitle
			}
		}
		
		resolve_title_and_vassal_change = scope:change_createNewTitle
		scope:migrator = { change_government = migrating_tribe_government }
	}
}
