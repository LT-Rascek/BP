#Scope is liege in question
can_have_wuzurgan_trigger = {
	scope:liege = { government_has_flag = government_uses_eranshar_authority }
	scope:liege = { is_independent_ruler = yes }
	scope:liege.primary_title.tier = tier_empire
}

#Scope is vassal in question
is_wuzurgan_trigger = {
	government_has_flag = government_uses_eranshar_authority
	OR = { #Have to be somewhat notable
		dynasty.dynasty_prestige_level >= 3
		is_powerful_vassal = yes
	}
	OR = { #Have to have a lot of land
		this.primary_title.tier >= tier_duchy
		this.domain_size >= 4
	}
}
