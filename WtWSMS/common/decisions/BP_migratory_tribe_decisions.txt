# Major Decisions
become_migratory_tribe_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/ep3_decision_laamp_neutral_mountain.dds"
	}
	sort_order = 82
	ai_check_interval = 120

	desc = become_migratory_tribe_decision_desc
	decision_group_type = major
	selection_tooltip = become_migratory_tribe_decision_tooltip

	is_shown = {
		#DLC check.
		has_ep3_dlc_trigger = yes
		primary_title.tier >= tier_county
		OR = {
			ai_can_valid_to_create_laamp_trigger = yes
			is_ai = no
		}
		#Standard filter checks.
		able_to_migrate_base_requirements_government_trigger = yes
		# Relevant cultures/faiths
		able_to_migrate_base_requirements_trigger = yes
		save_temporary_scope_as = potential_laamp_temp
		can_be_combatant_based_on_gender_trigger = { ARMY_OWNER = scope:potential_laamp_temp } # Registered fighting license
	}

	is_valid = {
		is_valid_for_tribal_migration_trigger = yes
		# Is not HOF
		custom_tooltip = {
			text = become_migratory_tribe_decision_hof_tt
			NOT = { faith.religious_head_title.holder = root }
		}
		# Relevant cultures/faiths
		custom_tooltip = {
			text = become_migratory_tribe_culture_religion_requirements_tt
			able_to_migrate_base_requirements_culture_religion_trigger = yes
		}
		become_migrating_tribe_title_and_culture_trigger = yes
	}

	cost = {
		#prestige = root.migrating_tribe_prestige_cost
		prestige = {
			value = 100
			#Size of the whole realm, weighted by culture
			every_held_title = {
				limit = { tier = tier_county }
				save_temporary_scope_as = the_county
				add = {
					value = 10
					multiply = {
						value = 1
						multiply = {
							value = 1
							if = {
								limit = {
									root.faith = scope:the_county.faith
									root.culture = scope:the_county.culture
								}
								add = 2
							}
							else_if = {
								limit = {
									OR = {
										root.faith = scope:the_county.faith
										root.culture = scope:the_county.culture
									}
								}
								add = 2
							}
							else = { subtract = 0 } #Zero out counties which aren't of the faith or culture
						}
					}
				}
			}
			#Bigger the rank, the more costly it is
			if = {
				limit = { root.primary_title.tier = tier_county }
				multiply = 1
			}
			else_if = {
				limit = { root.primary_title.tier = tier_duchy }
				multiply = 2.5
			}
			else = { multiply = 5 }
			#Historical mega-migrators do it cheaper
			if = {
				limit = { root.culture = { has_cultural_parameter = more_likely_to_create_migrating_tribes } }
				multiply = 0.5
			}
			#If not independent, it's cheaper
			if = {
				limit = { is_independent_ruler = no }
				multiply = 0.5
			}
			#If we are a legendary leader, it's very cheap
			if = {
				limit = { has_trait_with_flag = much_cheaper_migration_cost }
				multiply = 0.25
			}
			
			#Being more famous makes things cheaper
			divide = root.prestige_level
			min = 0
			max = 2500
		}
	}

	is_valid_showing_failures_only = {
		is_at_war = no
		is_alive = yes
		is_available_adult = yes
		has_contagious_deadly_disease_trigger = no
	}

	effect = {
		save_scope_as = migrator
		custom_tooltip = become_migratory_tribe_decision_title_created_tt
		hidden_effect_new_object = {
			create_character = {
				template = laamp_inheritor_template
				culture = root.capital_county.culture
				faith = root.capital_county.faith
				dynasty = generate
				location = root.capital_province
				after_creation = {
					save_scope_as = land_inheritor
					add_character_flag = new_laamp_inheritor
				}
			}
		}
		create_migratory_tribe_title_tooltip_effect = yes
		custom_tooltip = become_migratory_tribe_denude_areas_tt
		custom_tooltip = become_migratory_tribe_gold_tt
		custom_tooltip = become_migratory_tribe_armies_tt
		
		custom_tooltip = become_migratory_tribe_buildings_in_camp_tt
		
		trigger_event = { id = migratory_tribe_events.0100 }
		# Debug tracking
		if = {
			limit = {
				is_ai = yes
				debug_only = yes
			}
			if = {
				limit = { exists = global_var:ai_voluntary_laamp_count }
				change_global_variable = {
					name = ai_voluntary_laamp_count
					add = 1
				}
			}
			else = {
				set_global_variable = {
					name = ai_voluntary_laamp_count
					value = 1
				}
			}
			debug_log = "AI became adventurer voluntarily"
			debug_log_scopes = yes
		}
		stress_impact = {
			craven = minor_stress_impact_gain
			content = minor_stress_impact_gain
			lazy = minor_stress_impact_gain
		}
	}

	ai_potential = { is_valid_for_tribal_migration_trigger = yes }

	ai_will_do = {
		base = -75
		modifier = { add = ep3_laamp_chance_score_value }
		modifier = { # If we don't have this, everyone will leave your realm constantly
			add = -20
			culture = { has_cultural_parameter = more_likely_to_create_migrating_tribes }
		}
		modifier = {
			factor = 0
			#Historical non-migrators should not migrate
			BP_historical_migrators_trigger = { TARGET_CULTURE = root.culture }
		}
	}
}

# Minor Decisions
gather_provisions_tribe_decision = {
	picture = { reference = "gfx/interface/illustrations/decisions/ep3_decision_laamp_neutral_mountain.dds" }
	desc = gather_provisions_decision_desc
	selection_tooltip = gather_provisions_decision_tooltip
	ai_check_interval = 36
	sort_order = 200
	cooldown = { years = 3 }
	decision_group_type = migratory_tribe_minor

	is_shown = {
		#DLC check.
		has_ep3_dlc_trigger = yes
		is_landless_migratory_tribe_trigger = yes
	}

	is_valid_showing_failures_only = {
		is_available_allow_travelling = yes
	}

	effect = {
		custom_tooltip = {
			text = gather_provisions_decision_effect_tt
			trigger_event = ep3_decisions_event.7000
		}
		close_view = {
			view = decisions
			player = root
		}
	}

	ai_potential = {
		government_has_flag = government_is_migratory_tribe
		domicile ?= { provisions < max_provisions }
	}

	ai_will_do = {
		base = 100
	}
}
